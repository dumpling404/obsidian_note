title: 页面不存在
---

页面已移动或删除，<span id="count">3</span> 秒后自动跳转首页...

[点击这里立即跳转 →](/)

<script>
  let count = 3;
  const el = document.getElementById('count');
  const timer = setInterval(() => {
    count--;
    el.textContent = count;
    if (count <= 0) {
      clearInterval(timer);
      window.location.href = '/obsidian_note/';
    }
  }, 1000);
</script>
